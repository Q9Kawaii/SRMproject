#include <stdio.h>
#include <stdlib.h>

int main() {
    int m, n, p, q, i, j, k;
    printf("Enter rows & cols of A: ");
    scanf("%d %d", &m, &n);
    printf("Enter rows & cols of B: ");
    scanf("%d %d", &p, &q);

    if (n != p) { printf("Not possible\n"); return 0; }

    int **A = malloc(m * sizeof(int*));
    int **B = malloc(p * sizeof(int*));
    int **C = malloc(m * sizeof(int*));

    for (i = 0; i < m; i++) A[i] = malloc(n * sizeof(int));
    for (i = 0; i < p; i++) B[i] = malloc(q * sizeof(int));
    for (i = 0; i < m; i++) C[i] = calloc(q, sizeof(int)); // calloc â†’ starts with 0

    printf("Enter A:\n");
    for (i = 0; i < m; i++) for (j = 0; j < n; j++) scanf("%d", &A[i][j]);

    printf("Enter B:\n");
    for (i = 0; i < p; i++) for (j = 0; j < q; j++) scanf("%d", &B[i][j]);

    for (i = 0; i < m; i++)
        for (j = 0; j < q; j++)
            for (k = 0; k < n; k++)
                C[i][j] += A[i][k] * B[k][j];

    printf("Result C:\n");
    for (i = 0; i < m; i++) {
        for (j = 0; j < q; j++) printf("%d ", C[i][j]);
        printf("\n");
    }

    return 0;
}
